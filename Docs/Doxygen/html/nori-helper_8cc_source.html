264 void
265 NrHelper::InitializeOperationBand(OperationBandInfo* band, uint8_t flags)
266 {
267  NS_LOG_FUNCTION(this);
268 
269  static std::unordered_map<BandwidthPartInfo::Scenario, InitPathLossFn, std::hash<int>>
270  initLookupTable{
271  {BandwidthPartInfo::RMa,
272  std::bind(&InitRma, std::placeholders::_1, std::placeholders::_2)},
273  {BandwidthPartInfo::RMa_LoS,
274  std::bind(&InitRma_LoS, std::placeholders::_1, std::placeholders::_2)},
275  {BandwidthPartInfo::RMa_nLoS,
276  std::bind(&InitRma_nLoS, std::placeholders::_1, std::placeholders::_2)},
277  {BandwidthPartInfo::UMa,
278  std::bind(&InitUma, std::placeholders::_1, std::placeholders::_2)},
279  {BandwidthPartInfo::UMa_LoS,
280  std::bind(&InitUma_LoS, std::placeholders::_1, std::placeholders::_2)},
281  {BandwidthPartInfo::UMa_nLoS,
282  std::bind(&InitUma_nLoS, std::placeholders::_1, std::placeholders::_2)},
283  {BandwidthPartInfo::UMi_StreetCanyon,
284  std::bind(&InitUmi, std::placeholders::_1, std::placeholders::_2)},
285  {BandwidthPartInfo::UMi_StreetCanyon_LoS,
286  std::bind(&InitUmi_LoS, std::placeholders::_1, std::placeholders::_2)},
287  {BandwidthPartInfo::UMi_StreetCanyon_nLoS,
288  std::bind(&InitUmi_nLoS, std::placeholders::_1, std::placeholders::_2)},
289  {BandwidthPartInfo::InH_OfficeOpen,
290  std::bind(&InitIndoorOpen, std::placeholders::_1, std::placeholders::_2)},
291  {BandwidthPartInfo::InH_OfficeOpen_LoS,
292  std::bind(&InitIndoorOpen_LoS, std::placeholders::_1, std::placeholders::_2)},
293  {BandwidthPartInfo::InH_OfficeOpen_nLoS,
294  std::bind(&InitIndoorOpen_nLoS, std::placeholders::_1, std::placeholders::_2)},
295  {BandwidthPartInfo::InH_OfficeMixed,
296  std::bind(&InitIndoorMixed, std::placeholders::_1, std::placeholders::_2)},
297  {BandwidthPartInfo::InH_OfficeMixed_LoS,
298  std::bind(&InitIndoorMixed_LoS, std::placeholders::_1, std::placeholders::_2)},
299  {BandwidthPartInfo::InH_OfficeMixed_nLoS,
300  std::bind(&InitIndoorMixed_nLoS, std::placeholders::_1, std::placeholders::_2)},
301  {BandwidthPartInfo::UMa_Buildings,
302  std::bind(&InitUmaBuildings, std::placeholders::_1, std::placeholders::_2)},
303  {BandwidthPartInfo::UMi_Buildings,
304  std::bind(&InitUmiBuildings, std::placeholders::_1, std::placeholders::_2)},
305  {BandwidthPartInfo::V2V_Highway,
306  std::bind(&InitV2VHighway, std::placeholders::_1, std::placeholders::_2)},
307  {BandwidthPartInfo::V2V_Urban,
308  std::bind(&InitV2VUrban, std::placeholders::_1, std::placeholders::_2)},
309  };
310 
311  // Iterate over all CCs, and instantiate the channel and propagation model
312  for (const auto& cc : band->m_cc)
313  {
314  for (const auto& bwp : cc->m_bwp)
315  {
316  // Initialize the type ID of the factories by calling the relevant
317  // static function defined above and stored inside the lookup table
318  initLookupTable.at(bwp->m_scenario)(&m_pathlossModelFactory,
319  &m_channelConditionModelFactory);
320 
321  auto channelConditionModel =
322  m_channelConditionModelFactory.Create<ChannelConditionModel>();
323 
324  if (bwp->m_propagation == nullptr && flags & INIT_PROPAGATION)
325  {
326  bwp->m_propagation = m_pathlossModelFactory.Create<ThreeGppPropagationLossModel>();
327  bwp->m_propagation->SetAttributeFailSafe("Frequency",
328  DoubleValue(bwp->m_centralFrequency));
329  DynamicCast<ThreeGppPropagationLossModel>(bwp->m_propagation)
330  ->SetChannelConditionModel(channelConditionModel);
331  }
332 
333  if (bwp->m_3gppChannel == nullptr && flags & INIT_FADING)
334  {
335  bwp->m_3gppChannel =
336  m_spectrumPropagationFactory.Create<ThreeGppSpectrumPropagationLossModel>();
337  DynamicCast<ThreeGppSpectrumPropagationLossModel>(bwp->m_3gppChannel)
338  ->SetChannelModelAttribute("Frequency", DoubleValue(bwp->m_centralFrequency));
339  DynamicCast<ThreeGppSpectrumPropagationLossModel>(bwp->m_3gppChannel)
340  ->SetChannelModelAttribute("Scenario", StringValue(bwp->GetScenario()));
341  DynamicCast<ThreeGppSpectrumPropagationLossModel>(bwp->m_3gppChannel)
342  ->SetChannelModelAttribute("ChannelConditionModel",
343  PointerValue(channelConditionModel));
344  }
345 
346  if (bwp->m_channel == nullptr && flags & INIT_CHANNEL)
347  {
348  bwp->m_channel = m_channelFactory.Create<SpectrumChannel>();
349  bwp->m_channel->AddPropagationLossModel(bwp->m_propagation);
350  bwp->m_channel->AddPhasedArraySpectrumPropagationLossModel(bwp->m_3gppChannel);
351  }
352  }
353  }
