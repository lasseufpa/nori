301 void
302 MmWaveHelper::DoInitialize ()
303 {
304  NS_LOG_FUNCTION (this);
305 
306  // cc initialization
307  // if useCa=false and SetCcPhyParams() has not been called, setup a default CC.
308  if (!m_useCa && m_componentCarrierPhyParams.size () == 0)
309  {
310  NS_LOG_INFO ("useCa=false and empty CC map. Create the default CC.");
311 
312  // For custom configurations use the method SetCcPhyParams
313  Ptr<MmWavePhyMacCommon> phyMacConfig = CreateObject<MmWavePhyMacCommon> ();
314  Ptr<MmWaveComponentCarrier> cc = CreateObject<MmWaveComponentCarrier> ();
315  cc->SetConfigurationParameters (phyMacConfig);
316  cc->SetAsPrimary (true);
317 
318  //create the ccMap
319  std::map<uint8_t, MmWaveComponentCarrier > map;
320  map [0] = *cc;
321 
322  this->SetCcPhyParams (map);
323  }
324 
325  MmWaveChannelModelInitialization (); // channel initialization
326 
327  m_phyStats = CreateObject<MmWavePhyTrace> ();
328  m_radioBearerStatsConnector = CreateObject<MmWaveBearerStatsConnector> ();
329  m_enbStats = CreateObject<MmWaveMacTrace> ();
330 
331  // DoInitialize() will be called only once over the lifetime of an Object,
332  // just like DoDispose() is called only once.
333  m_basicCellId = m_cellIdCounter;
334 
335  // lte cc initialization
336  // if m_lteUseCa=false and SetLteCcPhyParams() has not been called, setup a default LTE CC
337  if (!m_lteUseCa && m_lteComponentCarrierPhyParams.size () == 0)
338  {
339  // create the map of LTE component carriers
340  Ptr<CcHelper> lteCcHelper = CreateObject<CcHelper> ();
341  lteCcHelper->SetNumberOfComponentCarriers (1);
342  lteCcHelper->SetUlEarfcn (18100);
343  lteCcHelper->SetDlEarfcn (100);
344  lteCcHelper->SetDlBandwidth (100);
345  lteCcHelper->SetUlBandwidth (100);
346  std::map< uint8_t, ComponentCarrier > lteCcMap = lteCcHelper->EquallySpacedCcs ();
347  lteCcMap.at (0).SetAsPrimary (true);
348  this->SetLteCcPhyParams (lteCcMap);
349  }
350 
351  LteChannelModelInitialization (); // lte channel initialization
352 
353  m_cnStats = 0; //core network stats calculator
354 
355  Object::DoInitialize ();
356 }

371 void
372 MmWaveHelper::MmWaveChannelModelInitialization (void)
373 {
374  NS_LOG_FUNCTION (this);
375  // setup of mmWave channel & related
376  //create a channel for each CC
377  for (std::map<uint8_t, MmWaveComponentCarrier >::iterator it = m_componentCarrierPhyParams.begin (); it != m_componentCarrierPhyParams.end (); ++it)
378  {
379  Ptr<SpectrumChannel> channel = m_channelFactory.Create<SpectrumChannel> ();
380  Ptr<MmWavePhyMacCommon> phyMacCommon = m_componentCarrierPhyParams.at (it->first).GetConfigurationParameters ();
381 
382  // create the channel condition model (if needed)
383  Ptr<ChannelConditionModel> ccm;
384  if (!m_channelConditionModelType.empty ())
385  {
386  ccm = m_channelConditionModelFactory.Create<ChannelConditionModel> ();
387  }
388 
389  // create the propagation loss model
390  if (!m_pathlossModelType.empty ())
391  {
392  Ptr<PropagationLossModel> plm = m_pathlossModelFactory.Create<PropagationLossModel> ();
393  if (plm)
394  {
395  plm->SetAttributeFailSafe ("Frequency", DoubleValue (phyMacCommon->GetCenterFrequency ()));
396 
397  // associate the channel condition model to the propagation loss model (if needed)
398  if (ccm)
399  {
400  plm->SetAttributeFailSafe ("ChannelConditionModel", PointerValue (ccm));
401  }
402 
403  // set the propagation loss model in the channel
404  channel->AddPropagationLossModel (plm);
405  }
406 
407  // store the propagation loss model
408  m_pathlossModel[it->first] = plm;
409  }
410  else
411  {
412  NS_LOG_WARN (this << " No PropagationLossModel!");
413  }
414 
415  // create and configure the SpectrumPropagationLossModel
416  if (!m_spectrumPropagationLossModelType.empty ())
417  {
418  Ptr<SpectrumPropagationLossModel> splm = m_spectrumPropagationLossModelFactory.Create<SpectrumPropagationLossModel> ();
419 
420  // if the selected model is ThreeGppSpectrumPropagationLossModel we
421  // need a special configuration procedure, otherwise, for the other
422  // models, we try to configure the frequency
423  Ptr<ThreeGppSpectrumPropagationLossModel> threeGppSplm = DynamicCast<ThreeGppSpectrumPropagationLossModel> (splm);
424  if (threeGppSplm)
425  {
426  threeGppSplm->SetChannelModelAttribute ("Frequency", DoubleValue (phyMacCommon->GetCenterFrequency ()));
427 
428  // the ThreeGppSpectrumPropagationLossModel must have the same ChannelConditionModel as the
429  // propagation loss model instace
430  if (ccm) // the channel condition model was created using the factory
431  {
432  threeGppSplm->SetChannelModelAttribute ("ChannelConditionModel", PointerValue (ccm));
433  }
434  else if (!m_pathlossModel.empty ()) // the channel condition model was created inside the propagation loss model
435  {
436  PointerValue ptr;
437  m_pathlossModel.at (it->first)->GetAttribute ("ChannelConditionModel", ptr);
438  ccm = ptr.Get<ChannelConditionModel> ();
439  threeGppSplm->SetChannelModelAttribute ("ChannelConditionModel", PointerValue (ccm));
440  }
441  else
442  {
443  NS_LOG_DEBUG ("ChannelConditionModel not set for ThreeGppSpectrumPropagationLossModel");
444  }
445  }
446  else
447  {
448  splm->SetAttributeFailSafe ("Frequency", DoubleValue (phyMacCommon->GetCenterFrequency ()));
449  }
450 
451  // set the propagation loss model in the channel
452  channel->AddSpectrumPropagationLossModel (splm);
453  }
454  else
455  {
456  NS_LOG_WARN (this << " No SpectrumPropagationLossModel!");
457  }
458 
459  m_channel [it->first] = channel;
460  } //end for
461 }

466  NS_LOG_FUNCTION (this);
467  // setup of LTE channels & related
468  m_downlinkChannel = m_lteChannelFactory.Create<SpectrumChannel> ();

470  m_downlinkPathlossModel = m_dlPathlossModelFactory.Create ();

472  if (dlSplm != 0)
473  {
474  NS_LOG_LOGIC (this << " using a SpectrumPropagationLossModel in DL");
475  m_downlinkChannel->AddSpectrumPropagationLossModel (dlSplm);
476  }
477  else
478  {
479  NS_LOG_LOGIC (this << " using a PropagationLossModel in DL");
480  Ptr<PropagationLossModel> dlPlm = m_downlinkPathlossModel->GetObject<PropagationLossModel> ();
481  NS_ASSERT_MSG (dlPlm != 0, " " << m_downlinkPathlossModel << " is neither PropagationLossModel nor SpectrumPropagationLossModel");
482  m_downlinkChannel->AddPropagationLossModel (dlPlm);
483  }
484 
485  m_uplinkPathlossModel = m_ulPathlossModelFactory.Create ();
486  Ptr<SpectrumPropagationLossModel> ulSplm = m_uplinkPathlossModel->GetObject<SpectrumPropagationLossModel> ();
