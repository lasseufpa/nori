234 void
235 NoriLteRlcSm::DoReceivePdu(LteMacSapUser::ReceivePduParameters rxPduParams)
236 {
237  NS_LOG_FUNCTION(this << rxPduParams.p);
238  // RLC Performance evaluation
239  RlcTag rlcTag;
240  Time delay;
241  bool ret = rxPduParams.p->FindFirstMatchingByteTag(rlcTag);
242  NS_ASSERT_MSG(ret, "RlcTag is missing");
243  delay = Simulator::Now() - rlcTag.GetSenderTimestamp();
244  NS_LOG_LOGIC(" RNTI=" << m_rnti << " LCID=" << (uint32_t)m_lcid << " size="
245  << rxPduParams.p->GetSize() << " delay=" << delay.As(Time::NS));
246  m_rxPdu(m_rnti, m_lcid, rxPduParams.p->GetSize(), delay.GetNanoSeconds());
247 }

249 void
250 NoriLteRlcSm::DoNotifyTxOpportunity(LteMacSapUser::TxOpportunityParameters txOpParams)
251 {
252  NS_LOG_FUNCTION(this << txOpParams.bytes);
253  LteMacSapProvider::TransmitPduParameters params;
254  RlcTag tag(Simulator::Now());
255 
256  params.pdu = Create<Packet>(txOpParams.bytes);
257  NS_ABORT_MSG_UNLESS(txOpParams.bytes > 0, "Bytes must be > 0");
258  /**
259  * For RLC SM, the packets are not passed to the upper layers, therefore,
260  * in the absence of an header we can safely byte tag the entire packet.
261  */
262  params.pdu->AddByteTag(tag, 1, params.pdu->GetSize());
263 
264  params.rnti = m_rnti;
265  params.lcid = m_lcid;
266  params.layer = txOpParams.layer;
267  params.harqProcessId = txOpParams.harqId;
268  params.componentCarrierId = txOpParams.componentCarrierId;
269 
270  // RLC Performance evaluation
271  NS_LOG_LOGIC(" RNTI=" << m_rnti << " LCID=" << (uint32_t)m_lcid
272  << " size=" << txOpParams.bytes);
273  m_txPdu(m_rnti, m_lcid, txOpParams.bytes);
274 
275  m_macSapProvider->TransmitPdu(params);
276  ReportBufferStatus();
277 }
